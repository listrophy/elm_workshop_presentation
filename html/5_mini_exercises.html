Undo

    initialModel : Model
    initialModel =
        []

Refactor: Msg


    type Msg
        = AddPoints Int
        | Undo

Refactor: View


    view : Model -> Html Msg
    view model =
        div
            []
            [ button [ onClick (AddPoints 1) ] [ text "free throw" ]
            , button [ onClick (AddPoints 2) ] [ text "jump shot" ]
            , button [ onClick (AddPoints 3) ] [ text "trey" ]
            , text (toString (List.sum model))
            , button [ onClick Undo ] [ text "undo!" ]
            ]

Refactor: Update


    update : Msg -> Model -> Model
    update msg model =
        case msg of
            AddPoints int ->
                int :: model
            Undo ->
                List.drop 1 model

Fresh Start: A Clock

A Clock

Let's move away from "beginnerProgram"

A Clock


    module Main exposing (main)

    import Html exposing (..)
    import Date
    import Time

    main =
        program
            { init = init
            , view = view
            , update = update
            , subscriptions = subscriptions
            }

Clock: Model & Init


    type alias Model = Maybe Time.Time

    init : (Model, Cmd Msg)
    init =
        ( Nothing
        , Cmd.none
        )

Clock: Msg


    type Msg
        = Tick Time.Time

Clock: View


view : Model -> Html Msg
view model =
    div [] <|
        case model of
            Just d ->
                [ text <| ":" ++ toString (Date.second d) ]

            Nothing ->
                [ text "don't know yet!" ]

Clock: Update


update : Msg -> Model -> (Model, Cmd Msg)
update msg model =
    case msg of
        Tick t ->
            (Just (Date.fromTime t), Cmd.none)

Clock: Subscriptions


subscriptions : Model -> Sub Msg
subscriptions model =
    Time.every (100 * Time.millisecond) Tick
